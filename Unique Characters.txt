/**
 *

public class JavaApplication6 {
 
    /**
     * @param str
     * @param args the command line arguments
     * @return
     */
   
    public static boolean hasUniqueChars(String str) {
        // TODO
        HashSet<Character> hs = new HashSet();
       
        for(int i =0; i<str.length(); i++) {
            if(!hs.add(str.charAt(i))) {
                return false;
            }
            hs.add(str.charAt(i));
        }
        return true;
     }
