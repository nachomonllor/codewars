  public static bool Narcissistic(int value)
        {

            string s = value + "";

            int sum = 0;
            for (int i = 0; i < s.Length; i++)
            {
                sum += (int) Math.Pow((s[i] - '0'), s.Length);
            }

            return sum == value;


        }
