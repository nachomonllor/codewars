using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;


public class SuffPref
{
    public static int solve(string s){  
         int max = 0;
           
            for (int i = 0; i <= s.Length / 2; i++)
            {
                if (s.Substring(0, i) == s.Substring(s.Length - i ))
                {
                    max = i ;
                }
              

            }

            return max;
    }
}
